94  private void process_operator(Token t)
95  {
96  Token A = null, B = null, R = null;
97 
98  switch (t.get_type())
99  {
100  //processing operators and functions, which needs two values from value stack
101  case Token.PLUS:
102  case Token.MINUS:
103  case Token.TIMES:
104  case Token.DIVISION:
105  case Token.POWER:
106  case Token.ROOT:
107  {
108  if (values_stack.is_empty())
109  {
110  throw new IllegalStateException();
111  }
112  else
113  {
114  B = values_stack.top();
115  values_stack.pop();
116  }
117 
118  if (values_stack.is_empty())
119  {
120  throw new IllegalStateException();
121  }
122  else
123  {
124  A = values_stack.top();
125  values_stack.pop();
126  }
127 
128  //processing root operator, it needs two values from value stack but in different order
129  if (t.get_type() == Token.ROOT)
130  {
131  R = t.operate(B.get_value(), A.get_value());
132  }
133  else
134  {
135  R = t.operate(A.get_value(), B.get_value());
136  }
137 
138  //pushing result to value stack
139  values_stack.push(R);
140  break;
141  }
142  //processing operators and functions, which needs only one value
143  case Token.ABSOLUTE_VALUE:
144  case Token.FACTORIAL:
145  {
146  if (values_stack.is_empty())
147  {
148  throw new IllegalStateException();
149  }
150  else
151  {
152  A = values_stack.top();
153  values_stack.pop();
154  }
155 
156  R = t.operate(A.get_value());
157 
158  values_stack.push(R);
159  break;
160  }
161 
162  } // end of switch

351  public void precedence_analysis(Token[] tokens)
352  {
353  //going through all tokens in token array and evaluating.
354  for (int n = 0; n < tokens.length; n++)
355  {
356  Token next_token = tokens[n];
357 
358  switch (next_token.get_type())
359  {
360  //if token is number it is pushed to value stack
361  case Token.NUMBER:
362  values_stack.push(next_token);
363  break;
364 
365  //if token is answer it is pushed to value stack as result of last expression(on start result is initialized on zero)
366  case Token.ANS:
367  values_stack.push(result);
368  break;
369 
370  //if token is left parenthesis it is pushed to operators stack
371  case Token.LEFT_PARENTHESIS:
372  operators_stack.push(next_token);
373  break;
374 
375  // pushing operators to operators stack instantly or processing them ,it depends on next conditions
376  case Token.PLUS:
377  case Token.MINUS:
378  case Token.TIMES:
379  case Token.DIVISION:
380  case Token.POWER:
381  case Token.ROOT:
382  case Token.ABSOLUTE_VALUE:
383  case Token.FACTORIAL:
384  {
385  // pushing operator to operators stack if condition is true
386  if (operators_stack.is_empty() || next_token.get_precedence() > operators_stack.top().get_precedence())
387  {
388  operators_stack.push(next_token);
389  }
390 
391  else
392  {
393  //processing operators until condition is true and then we can push next_token to operators stack
394  while (!operators_stack.is_empty() && next_token.get_precedence() <= operators_stack.top().get_precedence())
395  {
396  Token to_process = operators_stack.top();
397  operators_stack.pop();
398  try
399  {
400  process_operator(to_process);
401  }
402  catch (IllegalStateException e)
403  {
404  throw new IllegalStateException("Precedence analysis error.");
405  }
406  }
407  operators_stack.push(next_token);
408  }
409 
410  break;
411  }
412 
413  //if token is right parenthesis then operators in stack before left parenthesis need to be processed
414  case Token.RIGHT_PARENTHESIS:
415  {
416  //operators in parenthesis need to be processed until we hit on left parenthesis
417  while (!operators_stack.is_empty() && (operators_stack.top().get_type() == Token.PLUS || operators_stack.top().get_type() == Token.MINUS || operators_stack.top().get_type() == Token.TIMES || operators_stack.top().get_type() == Token.DIVISION || operators_stack.top().get_type() == Token.POWER || operators_stack.top().get_type() == Token.ROOT || operators_stack.top().get_type() == Token.ABSOLUTE_VALUE|| operators_stack.top().get_type() == Token.FACTORIAL))
418  {
419  Token to_process = operators_stack.top();
420  operators_stack.pop();
421  try
422  {
423  process_operator(to_process);
424  }
425  catch (IllegalStateException e)
426  {
427  throw new IllegalStateException("Precedence analysis error.");
428  }
429  }
430 
431  //popping out left parenthesis
432  if (!operators_stack.is_empty() && operators_stack.top().get_type() == Token.LEFT_PARENTHESIS)
433  {
434  operators_stack.pop();
435  }
436  else
437  {
438  throw new IllegalStateException("Precedence analysis error.");
439  }
440 
441  break;
442  }
443 
444  } // end of switch
445 
446  } // end of for cycle which goes through all of token from input
447 
448  // if we have still some operators in operators stack we need to process them until operators stack is empty
449  while (!operators_stack.is_empty() && (operators_stack.top().get_type() == Token.PLUS || operators_stack.top().get_type() == Token.MINUS || operators_stack.top().get_type() == Token.TIMES || operators_stack.top().get_type() == Token.DIVISION || operators_stack.top().get_type() == Token.POWER || operators_stack.top().get_type() == Token.ROOT || operators_stack.top().get_type() == Token.ABSOLUTE_VALUE || operators_stack.top().get_type() == Token.FACTORIAL))
450  {
